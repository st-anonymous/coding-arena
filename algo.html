  • MergeShot Algorithm

    
#include <bits/stdc++.h>
using namespace std;
void merge(int arr[], int l, int m, int r){
    int i = l, j = m+1, k = 0;
    int temp[r-l+1];
    while(i<=m && j<=r){
        if(arr[i] > arr[j]){
            temp[k] = arr[j];
            j++;
            k++;
        }
        else{
            temp[k] = arr[i];
            i++;
            k++;
        }
    }
    while(i<=m){
        temp[k] = arr[i];
        i++;
        k++;
    }
    while(j<=r){
        temp[k] = arr[j];
        j++;
        k++;
    }
    for(int i=l; i<=r; i++){
        arr[i] = temp[i-l];
    }
}

void mergeshot(int arr[], int l, int r){
    if(r <= l){
        return;
    }
    int m = (l+r)/2;
    mergeshot(arr, l, m);
    mergeshot(arr, m+1, r);
    merge(arr, l, m, r);
}

int main(){
    int n;
    cin >> n;
    int arr[n];
    int i = 0;
    while(i<n){
        cin >> arr[i];
        i++;
    }
    mergeshot(arr, 0, n-1);
    i = 0;
    while(i<n){
        cout << arr[i] << " ";
        i++;
    }
    return 0;
}

  • QuickSort Algorithm

    
#include <bits/stdc++.h>
using namespace std;
int pivot(int arr[], int l, int r){
    int i=l;
    int j=l;
    for(i; i<=r; i++){
        if(arr[i]<=arr[r]){
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
            j++;
        }
    }
    return j-1;
}
void quicksort(int arr[], int l, int r){
    if(l>=r){
        return;
    }
    int p = pivot(arr, l, r);
    quicksort(arr, l, p-1);
    quicksort(arr, p+1, r);
    return;
}
int main(){
    int n;
    cin >> n;
    int arr[n];
    int i=0;
    for(i; i<n; i++){
        cin >> arr[i];
    }
    quicksort(arr, 0, n-1);
    for(i=0; i<n; i++){
        cout << arr[i] << " ";
    }
    return 0;
}
